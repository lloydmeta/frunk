22/// Build a generic path that can be used for traversals
23#[proc_macro]
24pub fn path(input: TokenStream) -> TokenStream {
25    let expr = parse_macro_input!(input as Expr);
26    let path_type = build_path_type(expr);
27    let ast = quote! {
28        {
29            let p: #path_type = ::frunk_core::path::Path::new();
30            p
31        }
32    };
33    //    println!("ast: [{}]", ast);
34    TokenStream::from(ast)
35}

37/// Build a generic path that can be used for traversals
38#[proc_macro]
39#[allow(non_snake_case)]
40pub fn Path(input: TokenStream) -> TokenStream {
41    let expr = parse_macro_input!(input as Expr);
42    let path_type = build_path_type(expr);
43    let ast = quote! {
44        #path_type
45    };
46    //    println!("ast: [{}]", ast);
47    TokenStream::from(ast)
48}
