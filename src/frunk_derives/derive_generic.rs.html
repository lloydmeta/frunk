1use frunk_proc_macro_helpers::*;
2use proc_macro::TokenStream;
3use quote::ToTokens;
4use syn::Data;

6/// Given an AST, returns an implementation of Generic using HList
7///
8/// Only works with Structs and Tuple Structs
9pub fn impl_generic(input: TokenStream) -> impl ToTokens {
10    let ast = to_ast(input);
11    let name = &ast.ident;
12
13    let generics = &ast.generics;
14    let (impl_generics, ty_generics, where_clause) = generics.split_for_impl();
15
16    #[allow(clippy::let_and_return)]
17    let tree = match ast.data {
18        Data::Struct(ref data) => {
19            let field_bindings = FieldBindings::new(&data.fields);
20            let repr_type = field_bindings.build_hlist_type(FieldBinding::build_type);
21            let hcons_constr = field_bindings.build_hlist_constr(FieldBinding::build);
22            let type_constr = field_bindings.build_type_constr(FieldBinding::build);
23
24            quote! {
25                #[allow(non_snake_case, non_camel_case_types)]
26                impl #impl_generics ::frunk_core::generic::Generic for #name #ty_generics #where_clause {
27
28                    type Repr = #repr_type;
29
30                    fn into(self) -> Self::Repr {
31                        let #name #type_constr = self;
32                        #hcons_constr
33                    }
34
35                    fn from(r: Self::Repr) -> Self {
36                        let #hcons_constr = r;
37                        #name #type_constr
38                    }
39                }
40            }
41        }
42        _ => panic!("Only Structs are supported. Enums/Unions cannot be turned into Generics."),
43    };
44
45    //     print!("{}", tree);
46    tree
47}
