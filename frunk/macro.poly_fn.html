Returns a polymorphic function for use with mapping/folding heterogeneous types.

This macro is intended for use with simple scenarios, and doesnât handle trait implementation bounds or where clauses (it might in the future when procedural macros land). If it doesnât work for you, simply implement Func on your own.

use frunk_core::{Coprod, poly_fn};
type I32F32Str<'a> = Coprod!(i32, f32, &'a str);

let co1 = I32F32Str::inject("lollerskates");
let folded = co1.fold(poly_fn!(
  ['a] |x: &'a str| -> i8 { 1 },
  |x: i32| -> i8 { 2 },
  |f: f32| -> i8 { 3 },
));

assert_eq!(folded, 1);
